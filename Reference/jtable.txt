    public void loadDataPrefix(){ 
        DefaultTableModel dtmPrefix = new DefaultTableModel(); 
        dtmPrefix.addColumn("Id"); 
        dtmPrefix.addColumn("Name"); 
        dtmPrefix.addColumn("Gender"); 
        dtmPrefix.addColumn("Age"); 
        dtmPrefix.addColumn("Contact");
                
        try { 
            
            Connection con = DriverManager.getConnection("jdbc:ucanaccess://E://User//Project Work//DB.accdb");
            Statement stmt=con.createStatement(); 
            String sql = "select mid,mname,mgender,mage,mheight,mweight,mhair,meyes,missingdate,mcontact,lastseen from Missing";
            ResultSet rs=stmt.executeQuery(sql);
            int size = 0; 
            while(rs.next())
            { 
                size++; 
                dtmPrefix.addRow(new Object[]{rs.getInt(1), rs.getString(2), rs.getString(3), rs.getInt(4), rs.getString(5), rs.getInt(6),rs.getString(7),rs.getString(8),rs.getString(9),rs.getString(10),rs.getString(11),  }); 
                jTable1.setModel(dtmPrefix); 
            } 
        } 
        catch (SQLException preEr)
        { 
            JOptionPane.showMessageDialog(null, preEr,"Error  !",JOptionPane.WARNING_MESSAGE); 
        } 
    }
